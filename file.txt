from django.shortcuts import render, redirect
from django.http import JsonResponse
from django.views.decorators.http import require_POST
from django.contrib.auth.decorators import login_required
import json


@login_required
@require_POST
def preparer_remboursement(request):
    try:
        data = json.loads(request.body)

        # Sťcurisation : on ne fait jamais confiance aux donnťes JS
        client_id = data.get("client_id")
        montant = data.get("montant")
        reservation_id = data.get("reservation_id")

        if not client_id or not montant:
            return JsonResponse({"error": "Donnťes invalides"}, status=400)

        # Stockage en session sťcurisť
        request.session["remboursement_data"] = {
            "client_id": client_id,
            "montant": montant,
            
        }

        return JsonResponse({
            "success": True,
            "redirect_url": "/contrat/caisse/remboursement/"
        })

    except Exception as e:
        return JsonResponse({"error": str(e)}, status=500)

$("#btnRembourser").click(function () {

    let client_id = $("#client_id").val();
    let montant = $("#montant").val();
    let reservation_id = $("#reservation_id").val();

    $.ajax({
        url: "{% url 'preparer_remboursement' %}",
        type: "POST",
        headers: {
            "X-CSRFToken": "{{ csrf_token }}"
        },
        contentType: "application/json",
        data: JSON.stringify({
            client_id: client_id,
            montant: montant,
            reservation_id: reservation_id
        }),
        success: function (response) {
            if (response.success) {
                window.location.href = response.redirect_url;
            }
        },
        error: function (xhr) {
            alert("Erreur : " + xhr.responseJSON.error);
        }
    });
});
</script>
