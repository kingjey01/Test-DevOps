from django.shortcuts import render, redirect
from django.http import JsonResponse
from django.views.decorators.http import require_POST
from django.contrib.auth.decorators import login_required
import json


@login_required
@require_POST
def preparer_remboursement(request):
    try:
        data = json.loads(request.body)

        # Sécurisation : on ne fait jamais confiance aux données JS
        client_id = data.get("client_id")
        montant = data.get("montant")
        reservation_id = data.get("reservation_id")

        if not client_id or not montant:
            return JsonResponse({"error": "Données invalides"}, status=400)

        # Stockage en session sécurisé
        request.session["remboursement_data"] = {
            "client_id": client_id,
            "montant": montant,
            "reservation_id": reservation_id,
        }

        return JsonResponse({
            "success": True,
            "redirect_url": "/contrat/caisse/remboursement/"
        })

    except Exception as e:
        return JsonResponse({"error": str(e)}, status=500)
